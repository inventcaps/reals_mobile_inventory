STEP 1: PUSH SA GITHUB

STEP 2: DEPLOY SA RAILWAY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Punta sa: https://railway.app
2. Login with GitHub
3. Click "New Project"
4. "Deploy from GitHub repo"
5. Choose: reals_mobile_inventory
6. Hintayin 2-3 minutes

STEP 3: ADD ENVIRONMENT VARIABLES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Sa Railway Dashboard → Variables → Add:

  SECRET_KEY=<generate mo muna>
  DEBUG=False
  ALLOWED_HOSTS=*.railway.app
  DB_NAME=postgres
  DB_USER=postgres
  DB_PASSWORD=Reals_db_123
  DB_HOST=db.rczsumkmhoxjaycvggzt.supabase.co
  DB_PORT=5432

Para sa SECRET_KEY, run mo to:
  python -c "import secrets; print(secrets.token_urlsafe(50))"
